#include "SLLInventory.h"

void createList(List &L){ L.head = NULL; }

void insertLast(List &L, Product P){
    Node* N = new Node{P, NULL};
    if(L.head == NULL) L.head = N;
    else{
        Node* t = L.head;
        while(t->next) t = t->next;
        t->next = N;
    }
}

void viewList(List L){
    Node* t = L.head;
    int i=1;
    while(t){
        float harga = t->info.HargaSatuan * (1 - t->info.DiskonPersen/100);
        cout << i++ << ". " << t->info.Nama << " = " << harga << endl;
        t = t->next;
    }
}

#include "DLLPlaylist.h"

void createList(ListDLL &L){ L.head=L.tail=NULL; }

void insertLast(ListDLL &L, Song S){
    NodeDLL* N=new NodeDLL{S,NULL,NULL};
    if(L.head==NULL) L.head=L.tail=N;
    else{
        L.tail->next=N;
        N->prev=L.tail;
        L.tail=N;
    }
}

void viewList(ListDLL L){
    NodeDLL* t=L.head;
    int i=1;
    while(t){
        float score=0.8*t->info.PlayCount + 20*t->info.Rating;
        cout<<i++<<". "<<t->info.Title<<" = "<<score<<endl;
        t=t->next;
    }
}

void createStack(Stack &S){ S.Top=-1; }

void Push(Stack &S, Mahasiswa M){
    if(S.Top < MAX-1){
        S.Top++;
        S.data[S.Top]=M;
    }
}

void View(Stack S){
    for(int i=0;i<=S.Top;i++){
        cout<<i+1<<". "<<S.data[i].Nama<<" "<<S.data[i].NIM<<endl;
    }
}

#include "QueuePengiriman.h"

void createQueue(Queue &Q){ Q.head=Q.tail=-1; }

void enQueue(Queue &Q, Paket P){
    if(Q.tail<MAXQ-1){
        if(Q.head==-1) Q.head=0;
        Q.tail++;
        Q.data[Q.tail]=P;
    }
}

void viewQueue(Queue Q){
    for(int i=Q.head;i<=Q.tail;i++){
        cout<<i+1<<". "<<Q.data[i].Resi<<" "<<Q.data[i].Nama<<" "<<Q.data[i].Berat<<"kg\n";
    }
}
